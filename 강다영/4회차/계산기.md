		private void calculate_Click(object sender, EventArgs e)
        {
            int leftNumber = int.Parse(leftValue.Text);
            int rightNumber = int.Parse(rightValue.Text);

            if (operation.Text == "+")
            {                
                result.Text = (leftNumber + rightNumber).ToString();
            }
            else if (operation.Text == "-")
            {
                result.Text = (leftNumber - rightNumber).ToString();
            }
            else if (operation.Text == "*")
            {
                result.Text = (leftNumber * rightNumber).ToString();
            }
            else if (operation.Text == "/")
            {
                result.Text = (leftNumber / rightNumber).ToString();
            }
        }
